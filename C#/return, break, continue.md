```c#
int Count = 100;
```

# return
```c#
for(int i = 0; i < Count; i++)
{
  if(i == 12)
    return;
   
   Console.WriteLine("i");
}
```

# break  
```c#
for(int i = 0; i < Count; i++)
{
  if(i == 12)
    break;
    
  Console.WriteLine("i");
}
```
# continue  
```c#
for(int i = 0; i < Count; i++)
{
  if(i == 12)
    continue;
    
  Console.WriteLine("i");
}
```
