#include <iostream>
#include <iomanip>

using namespace std;

void DalPangE();
void SetBoard();

const int Max = 100;
int Num[Max][Max];
int Count;

int main()
{	
	cout << "숫자를 입력하세요." << endl;
	cin >> Count;

	SetBoard();
	DalPangE();
}

void DalPangE()
{
	for (int i = 0; i < Count; i++)
	{
		for (int x = 0; x < Count; x++)
		{
			cout << setfill('0') << setw(2) << Num[i][x] << " ";
		}
		cout << endl;
	}
}

void SetBoard()
{
	int Number = 0;

	for (int y = 0; y < Count; y++)
	{
		for (int x = 0; x < Count; x++)
		{
			if (y == 0)
				Num[y][x] = Number;
			else if (x == Count)
				Num[y][x] = Number;
				continue;

			Number++;
		}
	}
}
