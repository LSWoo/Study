#include <iostream>
#include <iomanip>

using namespace std;

void DalPangE();
void SetBoard();
bool CanGo(int a, int b);

const int Max = 100;
int Num[Max][Max];
int Count;

int main()
{	
	cout << "숫자를 입력하세요." << endl;
	cin >> Count;

	SetBoard();
	DalPangE();
}

void DalPangE()
{
	for (int i = 0; i < Count; i++)
	{
		for (int x = 0; x < Count; x++)
		{
			cout << setfill('0') << setw(2) << Num[i][x] << " ";
		}
		cout << endl;
	}
}

enum Dir
{
	Right,
	Down,
	Left,
	Up
};

void SetBoard()
{
	int dir = Right;

	int y = 0;
	int x = 0;
	int num = 1;

	while (true)
	{
		int nextY;
		int nextX;

		Num[y][x] = num;

		if (num == Count * Count)
			break;

		switch (dir)
		{
		case Right:
			nextX = x + 1;
			nextY = y;
			break;
		case Down:
			nextX = x;
			nextY = y + 1;
			break;
		case Left:
			nextX = x - 1;
			nextY = y;
			break;
		case Up:
			nextX = x;
			nextY = y - 1;
			break;
		}

		if (CanGo(nextY, nextX))
		{
			y = nextY;
			x = nextX;
		} 
		else {
			switch (dir)
			{
			case Right:
				dir = Down;
				break;
			case Down:
				dir = Left;
				break;
			case Left:
				dir = Up;
				break;
			case Up:
				dir = Right;
				break;
			}
		}

	}
}
bool CanGo(int nextY, int nextX)
{
	if (nextY < 0 || nextY >= Count)
		return false;
	if (nextX < 0 || nextX >= Count)
		return false;
	if (Num[nextY][nextX] != 0)
		return false;

	return true;
}
