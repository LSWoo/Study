# 반복문

```Assembly
; HelloWorld 을 10 번 출력하기

    mov bl, 0
    
    LAVEL_GO:
    cmp bl, 10
    ;bl 과 10 을 비교한 결과가 jl jge 인지 구분 후 해당 라벨로 점프한다.
    jl LAVEL_TO
    jge LAVEL_END
    
    LAVEL_TO:
    PRINT_STRING msg
    PRINT_DEC 1, bl
    add bl, 1
    NEWLINE
    jmp LAVEL_GO
    
    LAVEL_END:
    PRINT_STRING end
    NEWLINE

section .data
msg db 'HelloWorld', 0x00
end db 'End', 0x00
```
