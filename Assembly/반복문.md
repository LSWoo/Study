# 반복문

```Assembly
; HelloWorld 을 10 번 출력하기

    mov bl, 0
    
    LAVEL_LOOP:
    cmp bl, 10
    
    jge LAVEL_END
    
    PRINT_STRING msg
    PRINT_DEC 1, bl
    add bl, 1
    NEWLINE
    jmp LAVEL_LOOP
    
    LAVEL_END:
    PRINT_STRING end
    NEWLINE

section .data
msg db 'HelloWorld', 0x00
end db 'End', 0x00
```
